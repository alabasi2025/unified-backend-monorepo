import { openDB, DBSchema } from 'idb';

interface TileDB extends DBSchema {
  tiles: {
    key: string;
    value: {
      url: string;
      blob: Blob;
      timestamp: number;
    };
  };
}

export class OfflineStorageService {
  private dbPromise = openDB<TileDB>('maps-tiles', 1, {
    upgrade(db) {
      db.createObjectStore('tiles');
    },
  });

  async saveTile(url: string, blob: Blob): Promise<void> {
    const db = await this.dbPromise;
    await db.put('tiles', {
      url,
      blob,
      timestamp: Date.now()
    }, url);
  }

  async getTile(url: string): Promise<Blob | undefined> {
    const db = await this.dbPromise;
    const tile = await db.get('tiles', url);
    return tile?.blob;
  }

  async clearOldTiles(maxAge: number): Promise<void> {
    const db = await this.dbPromise;
    const allKeys = await db.getAllKeys('tiles');
    const now = Date.now();
    
    for (const key of allKeys) {
      const tile = await db.get('tiles', key);
      if (tile && (now - tile.timestamp) > maxAge) {
        await db.delete('tiles', key);
      }
    }
  }
}

// services/offline-tile-layer.ts
import * as L from 'leaflet';
import { OfflineStorageService } from './offline-storage.service';

export class OfflineTileLayer extends L.TileLayer {
  private storage = new OfflineStorageService();

  createTile(coords: L.Coords, done: L.DoneCallback): HTMLElement {
    const tile = document.createElement('img');
    const url = this.getTileUrl(coords);

    // Try to load from cache first
    this.storage.getTile(url).then(blob => {
      if (blob) {
        tile.src = URL.createObjectURL(blob);
        done(null, tile);
      } else {
        // Load from network and cache
        this.loadAndCacheTile(url, tile, done);
      }
    });

    return tile;
  }

  private async loadAndCacheTile(
    url: string,
    tile: HTMLImageElement,
    done: L.DoneCallback
  ): Promise<void> {
    try {
      const response = await fetch(url);
      const blob = await response.blob();
      
      // Cache the tile
      await this.storage.saveTile(url, blob);
      
      // Display the tile
      tile.src = URL.createObjectURL(blob);
      done(null, tile);
    } catch (error) {
      done(error as Error, tile);
    }
  }
}
